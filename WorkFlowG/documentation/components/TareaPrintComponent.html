import { Component, OnInit, OnDestroy } from '@angular/core';
import { Tarea } from '../shared/tarea';
import { TareaService } from '../shared/tarea.service';
import { ActivatedRoute } from '@angular/router';

/**
 * Componente de Printado de Tarea. Muestra el nombre de la Tarea a
 * printar en otra ruta
 */
@Component({
  selector: 'wfg-tarea-print',
  templateUrl: './tarea-print.component.html',
  styleUrls: ['./tarea-print.component.css']
})
export class TareaPrintComponent implements OnInit, OnDestroy {
  tarea: Tarea;
  id: string;
  params: any;

  /**
   * Constructor de TareaPrint. Activa la ruta donde se ejecutar√° el componente
   * @param tareaService par√°metro referente al Servicio de Tarea
   * @param activatedRoute par√°metro referente a la ruta activa donde se ejecutar√° el componente
   */
  constructor(private tareaService: TareaService, private activatedRoute: ActivatedRoute) { }

  /**
   * Hook de Inicio de TareaPrint. Establece la ruta activa de navegaci√≥n y hace una petici√≥n del
   * nombre de la Tarea a partir de su Id
   */
  ngOnInit() {
    this.params = this.activatedRoute.params.subscribe(params => this.id = params['id']);
    this.tareaService.getTareaHttp(this.id).subscribe((tarea: Tarea) => {
      this.tarea = tarea;
    });
  }

  /*
  * Hook de Destrucci√≥n de TareaPrint
  */
  ngOnDestroy(){
    this.params.unsubscribe();
  }
}
